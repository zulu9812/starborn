# enlarge the habitat by 1 step.
decision_starborn_enlarge_hab = {
	owned_planets_only = yes
	enactment_time = 300
	resources = {
		category = decisions
		cost = {
			trigger = {
				planet_size = 1
			}
            energy = 250
            alloys = 250
		}
		cost = {
			trigger = {
				planet_size = 2
			}
            energy = 250
            alloys = 250
		}
		cost = {
			trigger = {
				planet_size = 3
			}
            energy = 250
            alloys = 250
		}
		cost = {
			trigger = {
				planet_size = 4
			}
            energy = 250
            alloys = 500
		}
		cost = {
			trigger = {
				planet_size = 5
			}
            energy = 250
            alloys = 500
		}
	}
	
	potential = {
        is_planet_class = pc_habitat
		planet_size < 6
		owner = { has_civic = civic_starborn }
	}

	effect = {
		custom_tooltip = decision_starborn_enlarge_hab_desc
		if = {
			limit = {
				planet_size < 6
			}
			change_planet_size = 1
			remove_deposit = d_starborn_refund_size_2
			remove_deposit = d_starborn_refund_size_3
			remove_deposit = d_starborn_refund_size_4

			if = {
				limit = {
					planet_size = 3
				}
				add_deposit = d_starborn_refund_size_3
			}

			if = {
				limit = {
					planet_size = 4
				}
				add_deposit = d_starborn_refund_size_4
			}
		}
		else = {
			owner = {
				country_event = { id = starborn.2 }
			}
		}		
	}
	
	# Same AI waiting as Mastery of nature, wonder if the AI ever uses that.
	ai_weight = {
		weight = 1
		modifier = {
			factor = 0
			free_district_slots > 0
		}
	}
}
