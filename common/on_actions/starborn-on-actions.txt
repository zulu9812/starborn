# Triggers when the game starts
on_game_start = {
	events = {
		starborn_flags.1 # set global flag for patches
		starborn_flags.2 # set global flag if there is a starborn player
	}
}

on_game_start_country = {
	events = {
		starborn_init.1 # spawn starting habitats in system
		starborn_init.3 # general blurb
		starborn_init.7 # if using the OP system start, trigger living metal megaconsturction.
	}
}

empire_init_capital_planet = {
	events = {
		starborn_init.8 # nuke starting world and turn it into a tomb world: Paradise Lost
	}
}


# this = country
on_yearly_pulse_country = {
	events = {
		starborn.21 #check for completed city worlds
	}
}

# A planet has been colonized.
# Scope = Planet
on_colonized = {
	events = {
		starborn.10 #add refund deposits to small habitats
		starborn.11 # remove whole planet farming
	}
}

# Planet has been terraformed
# This = Planet
# From = Terraforming country
on_terraforming_complete = {
	events = {
		starborn.12 # remove scavenger districts and add farming when terraforming
		starborn.14 # remove scavenger districts and add farming  when terraforming homeworld
	}
}
